```typescript
import { describe, it, expect, beforeEach } from 'vitest';
import jwt from 'jsonwebtoken';
import { HttpRequest } from '@azure/functions';
import { requireAuth, getAuthContext } from '../middleware/auth.js';

const TEST_SECRET = 'test-jwt-secret';

function makeRequest(authHeader?: string): HttpRequest {
  return new HttpRequest({
    method: 'GET',
    url: 'http://localhost/api/test',
    headers: authHeader ? { authorization: authHeader } : {},
  });
}

function makeValidToken(overrides: Partial<{ sub: string; email: string; isAdmin: boolean; expiresIn: string | number }> = {}): string {
  const { sub = 'user123', email = 'test@example.com', isAdmin = false, expiresIn = '15m' } = overrides;
  return jwt.sign({ sub, email, isAdmin }, TEST_SECRET, { expiresIn } as any);
}

describe('requireAuth', () => {
  beforeEach(() => {
    process.env.JWT_SECRET = TEST_SECRET;
  });

  it('Test 1: returns null for valid JWT in Bearer header', async () => {
    const token = makeValidToken();
    const req = makeRequest(`Bearer ${token}`);
    const result = await requireAuth(req);
    expect(result).toBeNull();
  });

  it('Test 2: returns 401 when Authorization header is missing', async () => {
    const req = makeRequest();
    const result = await requireAuth(req);
    expect(result?.status).toBe(401);
  });

  it('Test 3: returns 401 when header does not start with Bearer', async () => {
    const token = makeValidToken();
    const req = makeRequest(`Basic ${token}`);
    const result = await requireAuth(req);
    expect(result?.status).toBe(401);
  });

  it('Test 4: returns 401 for expired token', async () => {
    // expiresIn: 1 millisecond → expired by the time jwt.verify runs
    const token = jwt.sign({ sub: 'u1', email: 'a@b.com', isAdmin: false }, TEST_SECRET, { expiresIn: '1ms' });
    await new Promise(resolve => setTimeout(resolve, 10)); // wait for expiry
    const req = makeRequest(`Bearer ${token}`);
    const result = await requireAuth(req);
    expect(result?.status).toBe(401);
  });

  it('Test 5: returns 401 for token signed with wrong secret', async () => {
    const token = jwt.sign({ sub: 'u1', email: 'a@b.com', isAdmin: false }, 'wrong-secret', { expiresIn: '15m' });
    const req = makeRequest(`Bearer ${token}`);
    const result = await requireAuth(req);
    expect(result?.status).toBe(401);
  });

  it('Test 6: getAuthContext returns userId/email/isAdmin after successful auth', async () => {
    const token = makeValidToken({ sub: 'abc123', email: 'coach@run.com', isAdmin: true });
    const req = makeRequest(`Bearer ${token}`);
    await requireAuth(req);
    const ctx = getAuthContext(req);
    expect(ctx.userId).toBe('abc123');
    expect(ctx.email).toBe('coach@run.com');
    expect(ctx.isAdmin).toBe(true);
  });

  it('Test 7: getAuthContext throws when requireAuth was not called first', () => {
    const req = makeRequest('Bearer some-token');
    expect(() => getAuthContext(req)).toThrow();
  });
});
```

```typescript
async function makeLoginRequest(body: object): Promise<HttpRequest> {
  return new HttpRequest({
    method: 'POST',
    url: 'http://localhost/api/auth/login',
    headers: { 'content-type': 'application/json' },
    body: { string: JSON.stringify(body) },
  });
}

async function makeRefreshRequest(body: object): Promise<HttpRequest> {
  return new HttpRequest({
    method: 'POST',
    url: 'http://localhost/api/auth/refresh',
    headers: { 'content-type': 'application/json' },
    body: { string: JSON.stringify(body) },
  });
}

async function makeLogoutRequest(body: object, authHeader?: string): Promise<HttpRequest> {
  return new HttpRequest({
    method: 'POST',
    url: 'http://localhost/api/auth/logout',
    headers: {
      'content-type': 'application/json',
      ...(authHeader ? { authorization: authHeader } : {}),
    },
    body: { string: JSON.stringify(body) },
  });
}

describe('POST /api/auth/login', () => {
  const PASSWORD = 'hunter2';
  let passwordHash: string;

  beforeEach(async () => {
    vi.clearAllMocks();
    process.env.JWT_SECRET = TEST_SECRET;
    passwordHash = await bcrypt.hash(PASSWORD, 10);
    mockUpdateOne.mockResolvedValue({});
    mockInsertOne.mockResolvedValue({ insertedId: 'token-doc-id' });
  });

  it('returns 200 with token, refreshToken, expiresIn:900 for valid credentials', async () => {
    mockFindOne
      .mockResolvedValueOnce({ _id: { toString: () => 'user1', id: 'user1' }, email: 'a@b.com', passwordHash, isAdmin: false })
      .mockResolvedValueOnce(null); // refresh_tokens lookup (not used in login path)

    const handler = handlers.get('login')!;
    const res = await handler(await makeLoginRequest({ email: 'a@b.com', password: PASSWORD }), ctx);

    expect(res.status).toBe(200);
    expect(res.jsonBody.token).toBeDefined();
    expect(res.jsonBody.refreshToken).toBeDefined();
    expect(res.jsonBody.expiresIn).toBe(900);

    // Verify JWT is signed with correct secret and has right payload shape
    const decoded = jwt.verify(res.jsonBody.token, TEST_SECRET) as any;
    expect(decoded.email).toBe('a@b.com');
    expect(decoded.isAdmin).toBe(false);
  });

  it('returns 401 Invalid credentials for wrong password', async () => {

    expect(res.status).toBe(401);
    expect(res.jsonBody.error).toBe('Invalid credentials');
  });

    const handler = handlers.get('login')!;
    const res = await handler(await makeLoginRequest({ email: 'a@b.com', password: PASSWORD }), ctx);

describe('POST /api/auth/refresh', () => {
  beforeEach(() => {
    vi.clearAllMocks();
    process.env.JWT_SECRET = TEST_SECRET;

  it('returns 200 with token and expiresIn:900 for valid refresh token', async () => {
```
