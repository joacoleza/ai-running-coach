**Imports:**
```typescript
import { app, HttpRequest, HttpResponseInit, InvocationContext } from '@azure/functions';
import jwt from 'jsonwebtoken';
import bcrypt from 'bcrypt';
import crypto from 'node:crypto';
import { ObjectId } from 'mongodb';
import { getDb } from '../shared/db.js';
import { User, RefreshToken } from '../shared/types.js';
import { requireAuth, getAuthContext } from '../middleware/auth.js';
```

**Helper — SHA-256 hash:**
```typescript
function sha256(input: string): string {
  return crypto.createHash('sha256').update(input).digest('hex');
}
```

**Helper — sign access token:**
```typescript
function signAccessToken(user: User & { _id: ObjectId }): string {
  const secret = process.env.JWT_SECRET;
  if (!secret) throw new Error('JWT_SECRET environment variable is not set');
  return jwt.sign(
    { sub: user._id.toString(), email: user.email, isAdmin: user.isAdmin },
    secret,
    { expiresIn: '15m' }
  );
}
```

**POST /api/auth/login:**
- Route: `'auth/login'`, methods: `['POST']`, authLevel: `'anonymous'`
- Read body: `const { email, password } = await req.json() as { email?: string; password?: string }`
- Validate: if either missing, return `{ status: 400, jsonBody: { error: 'email and password are required' } }`
- Look up user: `db.collection<User>('users').findOne({ email: email.toLowerCase().trim() })`
- If no user OR bcrypt.compare fails: return `{ status: 401, jsonBody: { error: 'Invalid credentials' } }` — same message for both cases (no user enumeration)
- Generate raw refresh token: `crypto.randomBytes(64).toString('hex')`
- Store in refresh_tokens: `{ userId: user._id, tokenHash: sha256(rawRefreshToken), expiresAt: new Date(Date.now() + 30 * 24 * 60 * 60 * 1000) }`
- Update lastLoginAt: `db.collection('users').updateOne({ _id: user._id }, { $set: { lastLoginAt: new Date(), updatedAt: new Date() } })`
- Sign JWT: `signAccessToken(user as User & { _id: ObjectId })`
- Return: `{ status: 200, jsonBody: { token, refreshToken: rawRefreshToken, expiresIn: 900 } }`

**POST /api/auth/refresh:**
- Route: `'auth/refresh'`, methods: `['POST']`, authLevel: `'anonymous'`
- Read body: `const { refreshToken } = await req.json() as { refreshToken?: string }`
- If missing: return `{ status: 401, jsonBody: { error: 'Refresh token required' } }`
- Look up: `db.collection<RefreshToken>('refresh_tokens').findOne({ tokenHash: sha256(refreshToken) })`
- If not found OR `doc.expiresAt < new Date()`: return `{ status: 401, jsonBody: { error: 'Invalid or expired refresh token' } }`
- Look up user: `db.collection<User>('users').findOne({ _id: doc.userId })`
- If no user: return `{ status: 401, jsonBody: { error: 'User not found' } }`
- Sign new access token: `signAccessToken(user as User & { _id: ObjectId })`
- Return: `{ status: 200, jsonBody: { token, expiresIn: 900 } }`

**POST /api/auth/logout:**
- Route: `'auth/logout'`, methods: `['POST']`, authLevel: `'anonymous'`
- Run requireAuth first: `const authError = await requireAuth(req); if (authError) return authError;`
- Read body: `const { refreshToken } = await req.json() as { refreshToken?: string }`
- If refreshToken provided, delete matching document: `db.collection('refresh_tokens').deleteOne({ tokenHash: sha256(refreshToken) })`
- If no refreshToken in body, still return 204 (best-effort logout — client may not send it)
- Return: `{ status: 204 }`

**Error handling:** Wrap each handler body in try/catch; on unexpected error return `{ status: 500, jsonBody: { error: 'Internal server error' } }`.
