**Fetching linked runs for the plan:**
Add a `useEffect` in PlanView that runs when `plan?._id` changes:
```typescript
const [linkedRuns, setLinkedRuns] = useState<Map<string, Run>>(new Map());

useEffect(() => {
  if (!plan?._id) return;
  // Fetch runs linked to this plan
  fetch(`/api/runs?planId=${plan._id}&limit=500`, { headers: authHeaders() })
    .then(r => r.json() as Promise<{ runs: Run[] }>)
    .then(data => {
      const map = new Map<string, Run>();
      for (const run of data.runs) {
        if (run.weekNumber && run.dayLabel) {
          map.set(`${run.weekNumber}-${run.dayLabel}`, run);
        }
      }
      setLinkedRuns(map);
    })
    .catch(() => { /* silently ignore */ });
}, [plan?._id]);
```

**3. `handleGetFeedback` function:**
```typescript
const { sendMessage, messages } = useChatContext();
const [isRequestingFeedback, setIsRequestingFeedback] = useState(false);

const handleGetFeedback = async () => {
  setIsRequestingFeedback(true);
  openCoachPanel(); // dispatch 'open-coach-panel' event

  const message = `Please give me a progress assessment for my training plan. ` +
    `Review my completed runs, how I'm tracking toward my goal, and any adjustments you'd recommend. ` +
    `Keep it concise — 3-4 sentences.`;

  try {
    await sendMessage(message);
    // Save the response as progressFeedback on the plan
    const lastMsg = [...messages].reverse().find(m => m.role === 'assistant');
    if (lastMsg?.content) {
      await fetch('/api/plan', {
        method: 'PATCH',
        headers: {
          'Content-Type': 'application/json',
          'x-app-password': localStorage.getItem('app_password') ?? '',
        },
        body: JSON.stringify({ progressFeedback: lastMsg.content }),
      });
      // Refresh plan to show updated progressFeedback
      window.dispatchEvent(new Event('plan-updated'));
    }
  } catch {
    // Coach panel will show any error
  } finally {
    setIsRequestingFeedback(false);
  }
};
```
