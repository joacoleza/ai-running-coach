    ```typescript
    import Anthropic from '@anthropic-ai/sdk';
    import { Db, ObjectId } from 'mongodb';
    import { buildSystemPromptWithRuns } from './prompts.js';
    import type { Run } from './types.js';

    /**
     * Trigger post-run coaching feedback.
     * Streams a coaching response to the chat collection and stores:
     * - run.insights on the run document
     * - plan.latestInsight and plan.estimatedFinishTime on the active plan
     *
     * This runs fire-and-forget from the API handler (no await needed in caller).
     */
    export async function triggerPostRunCoaching(
      db: Db,
      run: Run & { _id: ObjectId },
      planId?: string
    ): Promise<void> {
      try {
        // 1. Fetch active plan and recent run history (last 10 runs)
        const activePlan = planId
          ? await db.collection('plans').findOne({ _id: new ObjectId(planId) })
          : await db.collection('plans').findOne({ status: { $in: ['active', 'onboarding'] } });

        const recentRuns = await db.collection('runs')
          .find({})
          .sort({ date: -1 })
          .limit(10)
          .toArray();

        // 2. Build the post-run prompt message
        const runDate = run.date;
        const plannedDay = activePlan?.phases
          ?.flatMap((p: Record<string, unknown>) => (p.weeks as Record<string, unknown>[]) || [])
          ?.flatMap((w: Record<string, unknown>) => (w.days as Record<string, unknown>[]) || [])
          ?.find((d: Record<string, unknown>) => d.date === runDate);

        const userMessage = buildPostRunMessage(run, plannedDay as Record<string, unknown> | undefined);

        // 3. Build system prompt with run history context
        const systemPrompt = buildSystemPromptWithRuns({
          plan: activePlan,
          recentRuns: recentRuns as Run[],
        });

        // 4. Call Claude API (non-streaming — background process collects full response)
        const client = new Anthropic();
        let fullResponse = '';
        const stream = await client.messages.stream({
          model: 'claude-opus-4-6',
          max_tokens: 1024,
          system: systemPrompt,
          messages: [{ role: 'user', content: userMessage }],
        });

        for await (const chunk of stream) {
          if (chunk.type === 'content_block_delta' && chunk.delta.type === 'text_delta') {
            fullResponse += chunk.delta.text;
          }
        }

        // 5. Extract insights from response (strip <plan:update> tags for storage)
        const insightsText = fullResponse
          .replace(/<plan:update[^>]*>[\s\S]*?<\/plan:update>/g, '')
          .replace(/<plan:add[^>]*>[\s\S]*?<\/plan:add>/g, '')
          .trim();

        // 6. Extract estimated finish time if mentioned (look for pattern like "~2h05m" or "2:05")
        const finishMatch = insightsText.match(/~?(\d+h\d+m|\d+:\d+)/i);
        const estimatedFinishTime = finishMatch ? finishMatch[0] : undefined;

        // 7. Store coaching note on run record
        await db.collection('runs').updateOne(
          { _id: run._id },
          { $set: { insights: insightsText } }
        );

        // 8. Store latestInsight and estimatedFinishTime on active plan
        if (activePlan) {
          const planUpdate: Record<string, unknown> = { latestInsight: insightsText };
          if (estimatedFinishTime) planUpdate.estimatedFinishTime = estimatedFinishTime;
          await db.collection('plans').updateOne(
            { _id: activePlan._id },
            { $set: planUpdate }
          );
        }

        // 9. Append the coaching message to chat history (so it appears in CoachPanel)
        await db.collection('chats').insertOne({
          role: 'assistant',
          content: fullResponse,
          timestamp: new Date(),
          source: 'post-run-coaching',
          runId: run._id.toHexString(),
        });

        // 10. Apply <plan:update> tags from response (reuse existing plan day update logic)
        // Note: plan:update tags are applied client-side via useChat.ts when the message
        // is fetched from the chat endpoint. The raw response (with tags) is stored in the
        // chat collection so useChat can process them on next load.

      } catch (err) {
        // Post-run coaching is best-effort — log error but don't fail the run creation
        console.error('Post-run coaching failed:', err);
      }
    }

    function buildPostRunMessage(run: Run, plannedDay?: Record<string, unknown>): string {
      const lines: string[] = [
        `I just logged a run for ${run.date}:`,
        `- Distance: ${run.distance} km`,
        `- Time: ${run.duration} min`,
      ];
      if (run.pace) lines.push(`- Pace: ${run.pace}/km`);
      if (run.avgHR) lines.push(`- Avg HR: ${run.avgHR} bpm`);
      if (run.hrZoneLabel) lines.push(`- HR Zone: ${run.hrZoneLabel}`);
      if (run.elevationGain) lines.push(`- Elevation: ${run.elevationGain} m`);

      if (plannedDay) {
        lines.push('');
        lines.push(`The planned session for this day was:`);
        lines.push(`- Type: ${plannedDay.type}`);
        if (plannedDay.distance) lines.push(`- Planned distance: ${plannedDay.distance} km`);
        if (plannedDay.notes) lines.push(`- Notes: ${plannedDay.notes}`);
      } else {
        lines.push('');
        lines.push('This run was not linked to any planned session.');
      }

      lines.push('');
      lines.push('Please give me brief post-run feedback: how did the run compare to plan, one key insight, and whether any upcoming sessions should be adjusted. If you recommend plan changes, include them as <plan:update> tags.');

      return lines.join('\n');
    }
    ```
