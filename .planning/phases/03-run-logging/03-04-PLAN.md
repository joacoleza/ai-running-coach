    **1. Create `web/src/hooks/useRuns.ts`:**
    Follow the same pattern as `usePlan.ts`. Export a `useRuns()` hook with:

    ```typescript
    export interface RunData {
      _id: string;
      date: string;
      source: 'manual' | 'apple-health';
      distance: number;
      duration: number;
      avgHR?: number;
      maxHR?: number;
      pace?: string;
      cadence?: number;
      elevationGain?: number;
      hrZone?: number;
      hrZoneLabel?: string;
      planId?: string;
      planDayDate?: string;
      insights?: string;
      createdAt: string;
    }

    export function useRuns() {
      const [runs, setRuns] = useState<RunData[]>([]);
      const [loading, setLoading] = useState(false);
      const [error, setError] = useState<string | null>(null);

      const password = localStorage.getItem('app_password') || '';
      const headers = { 'x-app-password': password, 'Content-Type': 'application/json' };

      const fetchRuns = useCallback(async (filters?: {
        dateFrom?: string; dateTo?: string;
        distanceMin?: number; distanceMax?: number;
        durationMin?: number; durationMax?: number;
      }) => {
        setLoading(true);
        const params = new URLSearchParams();
        if (filters?.dateFrom) params.set('dateFrom', filters.dateFrom);
        if (filters?.dateTo) params.set('dateTo', filters.dateTo);
        if (filters?.distanceMin) params.set('distanceMin', String(filters.distanceMin));
        if (filters?.distanceMax) params.set('distanceMax', String(filters.distanceMax));
        if (filters?.durationMin) params.set('durationMin', String(filters.durationMin));
        if (filters?.durationMax) params.set('durationMax', String(filters.durationMax));
        const qs = params.toString();
        const res = await fetch(`/api/runs${qs ? '?' + qs : ''}`, { headers: { 'x-app-password': password } });
        if (res.ok) { setRuns(await res.json()); setError(null); }
        else { setError('Failed to fetch runs'); }
        setLoading(false);
      }, [password]);

      const createRun = useCallback(async (run: {
        date: string; source: 'manual'; distance: number; duration: number; avgHR?: number;
      }): Promise<RunData | null> => {
        const res = await fetch('/api/runs', { method: 'POST', headers, body: JSON.stringify(run) });
        if (res.ok) {
          const created = await res.json();
          // Dispatch plan-updated event so TrainingPlan page refreshes (day marked completed)
          window.dispatchEvent(new Event('plan-updated'));
          return created;
        }
        const err = await res.json().catch(() => ({ error: 'Failed to create run' }));
        throw new Error(err.error || 'Failed to create run');
      }, [password]);

      const deleteRun = useCallback(async (id: string): Promise<void> => {
        const res = await fetch(`/api/runs/${id}`, { method: 'DELETE', headers: { 'x-app-password': password } });
        if (!res.ok) {
          const err = await res.json().catch(() => ({ error: 'Failed to delete run' }));
          throw new Error(err.error || 'Failed to delete run');
        }
      }, [password]);

      const getRun = useCallback(async (id: string): Promise<RunData | null> => {
        const res = await fetch(`/api/runs/${id}`, { headers: { 'x-app-password': password } });
        if (res.ok) return await res.json();
        return null;
      }, [password]);

      return { runs, loading, error, fetchRuns, createRun, deleteRun, getRun };
    }
    ```
