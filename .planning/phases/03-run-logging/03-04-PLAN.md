// D-04: "Monday 03/04/2026" = day-of-week + DD/MM/YYYY
function formatRunDate(isoDate: string): string {
  const d = new Date(isoDate + 'T12:00:00');
  const weekday = d.toLocaleDateString('en-GB', { weekday: 'long' });
  const dd = String(d.getDate()).padStart(2, '0');
  const mm = String(d.getMonth() + 1).padStart(2, '0');
  return `${weekday} ${dd}/${mm}/${d.getFullYear()}`;
}

// Pace formatting: decimal minutes to "M:SS/km"
function formatPace(pace: number): string {
  const m = Math.floor(pace);
  const s = Math.round((pace - m) * 60);
  return `${m}:${String(s).padStart(2, '0')}/km`;
}

```typescript
const PAGE_SIZE = 20;

const loadRuns = useCallback(async (reset = false) => {
  if (reset) setOffset(0);
  const currentOffset = reset ? 0 : offset;
  setIsLoading(reset);
  if (!reset) setIsLoadingMore(true);
  try {
    const result = await fetchRuns({
      limit: PAGE_SIZE,
      offset: currentOffset,
      dateFrom: dateFrom || undefined,
      dateTo: dateTo || undefined,
      distanceMin: distanceMin ? parseFloat(distanceMin) : undefined,
      distanceMax: distanceMax ? parseFloat(distanceMax) : undefined,
    });
    setRuns(prev => reset ? result.runs : [...prev, ...result.runs]);
    setTotal(result.total);
    if (!reset) setOffset(currentOffset + result.runs.length);
  } catch (err) {
    setError(err instanceof Error ? err.message : 'Failed to load runs');
  } finally {
    setIsLoading(false);
    setIsLoadingMore(false);
  }
}, [offset, dateFrom, dateTo, distanceMin, distanceMax]);

// Load on mount and filter changes
useEffect(() => { void loadRuns(true); }, [dateFrom, dateTo, distanceMin, distanceMax]);
```

**Infinite scroll:** Use an IntersectionObserver on a sentinel div at the bottom:
```typescript
const sentinelRef = useRef<HTMLDivElement>(null);
useEffect(() => {
  const el = sentinelRef.current;
  if (!el) return;
  const obs = new IntersectionObserver(entries => {
    if (entries[0]?.isIntersecting && runs.length < total && !isLoadingMore) {
      void loadRuns(false);
    }
  });
  obs.observe(el);
  return () => obs.disconnect();
}, [runs.length, total, isLoadingMore, loadRuns]);
```
