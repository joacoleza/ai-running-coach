    ```typescript
      route: 'runs/sas-token',
      handler: async (req: HttpRequest, context: InvocationContext): Promise<HttpResponseInit> => {
        const denied = await requirePassword(req);
        if (denied) return denied;

        const accountName = process.env.BLOB_ACCOUNT_NAME || 'devstoreaccount1';
        const accountKey = process.env.BLOB_ACCOUNT_KEY || 'Eby8vdM02xNOcqFlqUwJPLlmEtlCDXJ1OUzFT50uSRZ6IFsuFq2UVErCz4I6tq/K1SZFPTOtr/KBHBeksoGMGw==';
        const containerName = 'apple-health-uploads';
        const blobName = `upload-${Date.now()}-${Math.random().toString(36).slice(2)}.zip`;

        // Create upload record in MongoDB
        const db = await getDb();
        const result = await db.collection('uploads').insertOne({
          status: 'pending',
          blobName,
          createdAt: new Date(),
          updatedAt: new Date(),
        });

        // Generate SAS token
        const sharedKeyCredential = new StorageSharedKeyCredential(accountName, accountKey);
        const sasToken = generateBlobSASQueryParameters({
          containerName,
          blobName,
          startsOn: new Date(),
          expiresOn: new Date(Date.now() + 15 * 60 * 1000), // 15 min
          permissions: BlobSASPermissions.parse('cw'),
        }, sharedKeyCredential).toString();

        // For local dev (Azurite), use the Azurite URL format
        const isAzurite = accountName === 'devstoreaccount1';
        const baseUrl = isAzurite
          ? `http://127.0.0.1:10000/${accountName}/${containerName}/${blobName}`
          : `https://${accountName}.blob.core.windows.net/${containerName}/${blobName}`;

        return {
          status: 200,
          jsonBody: {
            uploadId: result.insertedId.toHexString(),
            uploadUrl: `${baseUrl}?${sasToken}`,
            blobName,
          },
        };
      },
    });
    ```

    ```typescript
    import { app, HttpRequest, HttpResponseInit, InvocationContext } from '@azure/functions';
    import { ObjectId } from 'mongodb';
    import { requirePassword } from '../middleware/auth.js';
    import { getDb } from '../shared/db.js';

    app.http('runUploadStatus', {
      methods: ['GET'],
      authLevel: 'anonymous',
      route: 'runs/upload-status/{uploadId}',
      handler: async (req: HttpRequest, context: InvocationContext): Promise<HttpResponseInit> => {
        const denied = await requirePassword(req);
        if (denied) return denied;

        const uploadId = req.params['uploadId'];
        if (!uploadId || !ObjectId.isValid(uploadId)) {
          return { status: 400, jsonBody: { error: 'Invalid uploadId' } };
        }

        const db = await getDb();
        const upload = await db.collection('uploads').findOne({ _id: new ObjectId(uploadId) });
        if (!upload) return { status: 404, jsonBody: { error: 'Upload not found' } };

        return {
          status: 200,
          jsonBody: {
            status: upload.status,
            runsExtracted: upload.runsExtracted,
            errorMessage: upload.errorMessage,
          },
        };
      },
    });
    ```

    Create `api/src/functions/runBlobTrigger.ts`:
    ```typescript
    import { app, InvocationContext } from '@azure/functions';
    import { getDb } from '../shared/db.js';
    import { parseAppleHealthZip } from '../shared/runParser.js';
    import { computeHrZone } from '../shared/hrZones.js';
    import type { Run } from '../shared/types.js';

    app.storageBlob('parseAppleHealthUpload', {
      path: 'apple-health-uploads/{name}',
      connection: 'AzureWebJobsStorage',
      handler: async (blob: Buffer, context: InvocationContext) => {
        const blobName = context.triggerMetadata?.name as string;
        const db = await getDb();

        // 1. Find upload record, set status='processing'
        const uploadDoc = await db.collection('uploads').findOneAndUpdate(
          { blobName },
          { $set: { status: 'processing', updatedAt: new Date() } },
          { returnDocument: 'after' }
        );
        if (!uploadDoc) {
          context.error(`No upload record found for blob: ${blobName}`);
          return;
        }

        try {
          // 2. Parse ZIP
          const workouts = await parseAppleHealthZip(blob);

          // 3. Get user's maxHR from active plan goal (if set) for HR zone computation
          const activePlan = await db.collection('plans').findOne(
            { status: { $in: ['active', 'onboarding'] } },
            { projection: { 'goal.maxHR': 1 } }
          );
          const userMaxHR = activePlan?.goal?.maxHR;

          // 4. Create run records, skip duplicates
          let runsCreated = 0;
          for (const workout of workouts) {
            // Convert startDate to local YYYY-MM-DD
            // Apple Health format: "2026-03-15 07:00:00 +0000"
            const startDt = new Date(workout.startDate);
            const localDate = startDt.getFullYear() + '-' +
              String(startDt.getMonth() + 1).padStart(2, '0') + '-' +
              String(startDt.getDate()).padStart(2, '0');

            // Skip duplicates: check if run already exists for this date+source
            const existing = await db.collection('runs').findOne({
              date: localDate,
              source: 'apple-health',
            });
            if (existing) continue;

            // Compute pace
            let pace: string | undefined;
            if (workout.distance > 0) {
              const totalSec = workout.duration * 60;
              const paceSec = totalSec / workout.distance;
              const mm = Math.floor(paceSec / 60);
              const ss = Math.round(paceSec % 60);
              pace = `${mm}:${String(ss).padStart(2, '0')}`;
            }

            // Compute HR zone
            let hrZone: number | undefined;
            let hrZoneLabel: string | undefined;
            if (workout.avgHR && userMaxHR) {
              const zone = computeHrZone(workout.avgHR, userMaxHR);
              hrZone = zone.zone;
              hrZoneLabel = zone.label;
            }

            // Build run document
            const run: Omit<Run, '_id'> = {
              date: localDate,
              source: 'apple-health',
              distance: workout.distance,
              duration: workout.duration,
              avgHR: workout.avgHR,
              maxHR: workout.maxHR,
              pace,
              cadence: workout.cadence,
              elevationGain: workout.elevationGain,
              hrZone,
              hrZoneLabel,
              createdAt: new Date(),
            };

            // Run-to-plan linking: check active plan for matching day
            const activePlanFull = await db.collection('plans').findOne({
              status: { $in: ['active', 'onboarding'] },
            });
            if (activePlanFull) {
              // Find matching day in plan phases
              let dayFound = false;
              for (const phase of activePlanFull.phases || []) {
                for (const week of phase.weeks || []) {
                  for (const day of week.days || []) {
                    if (day.date === localDate) {
                      dayFound = true;
                      if (!day.completed) {
                        // Link run and mark day completed
                        (run as Record<string, unknown>).planId = activePlanFull._id!.toHexString();
                        (run as Record<string, unknown>).planDayDate = localDate;

                        // Mark plan day as completed
                        await db.collection('plans').updateOne(
                          { _id: activePlanFull._id },
                          { $set: { 'phases.$[p].weeks.$[w].days.$[d].completed': true, 'phases.$[p].weeks.$[w].days.$[d].skipped': false } },
                          { arrayFilters: [{ 'p.weeks': { $exists: true } }, { 'w.days': { $exists: true } }, { 'd.date': localDate }] }
                        );
                      }
                      // If already completed, store unlinked (don't error in blob trigger)
                    }
                  }
                  if (dayFound) break;
                }
                if (dayFound) break;
              }
            }

            await db.collection('runs').insertOne(run);
            runsCreated++;
          }

          // 5. Update upload status to done
          await db.collection('uploads').updateOne(
            { blobName },
            { $set: { status: 'done', runsExtracted: runsCreated, updatedAt: new Date() } }
          );

          context.log(`Parsed ${workouts.length} workouts, created ${runsCreated} runs from blob: ${blobName}`);
        } catch (err: unknown) {
          const msg = err instanceof Error ? err.message : String(err);
          context.error(`Failed to parse blob ${blobName}: ${msg}`);
          await db.collection('uploads').updateOne(
            { blobName },
            { $set: { status: 'error', errorMessage: msg, updatedAt: new Date() } }
          );
        }
      },
    });
    ```
