```typescript
export type FilterPreset =
  | 'current-plan'
  | 'last-4-weeks'
  | 'last-8-weeks'
  | 'last-3-months'
  | 'last-12-months'
  | 'this-year'
  | 'all-time'

export interface DashboardStats {
  totalDistance: string   // e.g. "42.5km"
  totalRuns: number
  totalTime: string       // e.g. "3h25m"
  adherence: string       // e.g. "75%" or "N/A"
}

export interface WeeklyDataPoint {
  weekLabel: string       // e.g. "Apr 7"
  distance: number
}

export interface PaceDataPoint {
  weekLabel: string
  pace: number            // min/km
}

export function useDashboard(): {
  activeFilter: FilterPreset
  setActiveFilter: (f: FilterPreset) => void
  stats: DashboardStats
  weeklyData: WeeklyDataPoint[]
  paceData: PaceDataPoint[]
  isLoading: boolean
  hasPlan: boolean
}
```

- XAxis, YAxis, CartesianGrid, Tooltip, ResponsiveContainer
