**PlanActions.tsx — Remove Update Plan button, make Archive compact:**

The current PlanActions renders two buttons: "Update Plan" and "Close & Archive". The Update Plan button is redundant (coach panel handles plan updates). Remove it entirely.

Replace the entire PlanActions component with a compact archive-only version that renders a small button suitable for placement inline with the title:

```tsx
interface PlanActionsProps {
  hasActivePlan: boolean;
  onArchive: () => void;
}

export function PlanActions({ hasActivePlan, onArchive }: PlanActionsProps) {
  const handleArchive = () => {
    if (window.confirm('Archive this plan? You can view it later in the Archive section.')) {
      onArchive();
    }
  };

  if (!hasActivePlan) return null;

  return (
    <button
      onClick={handleArchive}
      className="cursor-pointer px-2 py-1 rounded text-xs font-medium text-gray-500 border border-gray-300 hover:bg-red-50 hover:text-red-600 hover:border-red-300 transition-colors"
    >
      Archive
    </button>
  );
}
```

Remove the `onUpdate` prop entirely from the interface and component signature.
