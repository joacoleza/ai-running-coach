       ```typescript
       test.describe('Auth flow', () => {
         test.beforeEach(async ({ page }) => {
           // Clear localStorage to start unauthenticated
           await page.goto('/')
           await page.evaluate(() => localStorage.clear())
           await page.reload()
         })

         test('shows password page when not authenticated', async ({ page }) => {
           await page.goto('/')
           await expect(page.getByText('AI Running Coach')).toBeVisible()

         test('correct password grants access', async ({ page }) => {
           await page.goto('/')
           await page.getByPlaceholder('Enter password').fill('e2e-test-password')
           await page.getByRole('button', { name: 'Unlock' }).click()
           // After success, should see app content (Dashboard nav link)
           await expect(page.getByText('Dashboard')).toBeVisible({ timeout: 10_000 })
         })

         test('wrong password shows error', async ({ page }) => {
           await page.goto('/')
           await page.getByPlaceholder('Enter password').fill('wrong-password')
           await page.getByRole('button', { name: 'Unlock' }).click()
           await expect(page.getByText('Wrong password')).toBeVisible({ timeout: 10_000 })
           // Should still be on password page
           await expect(page.getByPlaceholder('Enter password')).toBeVisible()
         })

         test('logout returns to password page', async ({ page }) => {
           // First login
           await page.goto('/')
           await page.getByPlaceholder('Enter password').fill('e2e-test-password')
           await page.getByRole('button', { name: 'Unlock' }).click()
           await expect(page.getByText('Dashboard')).toBeVisible({ timeout: 10_000 })
           // Click logout
           await page.getByRole('button', { name: /Logout/i }).click()
           // Should be back on password page
           await expect(page.getByText('AI Running Coach')).toBeVisible({ timeout: 10_000 })
         })
       })
       ```

    5. **Create `e2e/lockout.spec.ts`**:
       ```typescript
       import { test, expect } from '@playwright/test'
       import { MongoClient } from 'mongodb'

       const MONGO_URI = process.env.MONGODB_CONNECTION_STRING || 'mongodb://localhost:27017'

       test.describe('Lockout behavior', () => {
         test.beforeEach(async () => {
           // Reset lockout state in MongoDB before each test
           const client = new MongoClient(MONGO_URI)
           await client.connect()
           await client.db('running-coach').collection('auth').deleteMany({})
           await client.close()
         })

         test('shows locked error after too many wrong attempts', async ({ page }) => {
           // Simulate 30 failed attempts directly in MongoDB
           // (Faster than submitting the form 30 times)
           const client = new MongoClient(MONGO_URI)
           await client.connect()
           await client.db('running-coach').collection('auth').insertOne({
             _id: 'lockout' as any,
             failureCount: 30,
             blocked: true,
           })
           await client.close()

           await page.goto('/')
           await page.evaluate(() => localStorage.clear())
           await page.reload()

           await page.getByPlaceholder('Enter password').fill('e2e-test-password')
           await page.getByRole('button', { name: 'Unlock' }).click()
           await expect(page.getByText('Service locked. Contact administrator.')).toBeVisible({ timeout: 10_000 })
         })

         test('wrong password increments failure count in DB', async ({ page }) => {
           await page.goto('/')
           await page.evaluate(() => localStorage.clear())
           await page.reload()

           await page.getByPlaceholder('Enter password').fill('wrong')
           await page.getByRole('button', { name: 'Unlock' }).click()
           await expect(page.getByText('Wrong password')).toBeVisible({ timeout: 10_000 })

           // Assert MongoDB state
           const client = new MongoClient(MONGO_URI)
           await client.connect()
           const doc = await client.db('running-coach').collection('auth').findOne({ _id: 'lockout' as any })
           expect(doc?.failureCount).toBe(1)
           expect(doc?.blocked).toBeFalsy()
           await client.close()
         })
       })
       ```
       NOTE: The lockout spec imports `mongodb` directly — it needs the mongodb driver installed. Add `mongodb` as a devDependency in root package.json: `npm install -D mongodb@^7.1.0`

    1. **Create `.github/workflows/ci.yml`**:

       IMPORTANT: `master` is branch-protected (see CLAUDE.md). The coverage badge action MUST NOT
       push directly to master. Instead, badges are pushed to an unprotected `badges` branch and
       referenced via raw.githubusercontent.com URL. Per RESEARCH.md Pitfall 6.

       ```yaml
       name: CI

       on:
         pull_request:
           branches: [master]
         push:
           branches: [master]

       jobs:
         test:
           runs-on: ubuntu-latest
           steps:
             - uses: actions/checkout@v4

             - uses: actions/setup-node@v5
               with:
                 node-version: '22'

             - name: Start MongoDB
               uses: supercharge/mongodb-github-action@1.12.1
               with:
                 mongodb-version: '7.0'

             - name: Install root deps
               run: npm ci

             - name: Install web deps
               run: cd web && npm ci

             - name: Install API deps
               run: cd api && npm ci

             - name: Run API tests
               run: cd api && npx vitest run
               env:
                 MONGODB_CONNECTION_STRING: mongodb://localhost:27017
                 APP_PASSWORD: ci-test-password

             - name: Run web tests with coverage
               run: cd web && npx vitest run --coverage

             - name: Install Playwright browsers
               run: npx playwright install --with-deps chromium

             - name: Run E2E tests
               run: npx playwright test
               env:
                 MONGODB_CONNECTION_STRING: mongodb://localhost:27017
                 APP_PASSWORD: ci-test-password
                 CI: true

         coverage-badges:
           runs-on: ubuntu-latest
           needs: test
           if: github.ref == 'refs/heads/master' && github.event_name == 'push'
           permissions:
             contents: write
           steps:
             - uses: actions/checkout@v4

             - uses: actions/setup-node@v5
               with:
                 node-version: '22'

             - name: Install web deps
               run: cd web && npm ci

             - name: Run web tests with coverage
               run: cd web && npx vitest run --coverage

             - name: Generate coverage badges
               uses: jpb06/coverage-badges-action@latest
               with:
                 coverage-summary-path: ./web/coverage/coverage-summary.json
                 output-folder: ./badges
                 no-commit: true

             - name: Push badges to badges branch
               run: |
                 git config user.name "github-actions[bot]"
                 git config user.email "github-actions[bot]@users.noreply.github.com"
                 # Create orphan badges branch content
                 mkdir -p /tmp/badges-deploy
                 cp -r ./badges/* /tmp/badges-deploy/
                 git fetch origin badges:badges 2>/dev/null || true
                 git checkout --orphan badges-tmp
                 git rm -rf .
                 mkdir badges
                 cp /tmp/badges-deploy/* badges/
                 git add badges/
                 git commit -m "Update coverage badges [skip ci]"
                 git push origin HEAD:badges --force
       ```

       Key design decisions:
       - Single `test` job runs API tests, web tests with coverage, and E2E tests sequentially (E2E needs the same MongoDB instance)
       - `coverage-badges` job runs only on push to master (not PRs)
       - Badge SVGs are pushed to an unprotected `badges` branch, avoiding master branch protection (per CLAUDE.md + RESEARCH.md Pitfall 6)
       - `jpb06/coverage-badges-action` generates the SVGs with `no-commit: true`; a separate step pushes to the `badges` branch
       - `supercharge/mongodb-github-action@1.12.1` starts MongoDB 7.0 with health check
       - E2E tests get `CI: true` env var so Playwright sets `reuseExistingServer: false`
       - `[skip ci]` in badge commit message prevents infinite CI loops
